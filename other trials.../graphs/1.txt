)

# Create the plot
plot <- ggplot(data, aes(x = Metric, y = Value, group = Algorithm, color = Algorithm)) +

  geom_point(size = 3) +
  labs(title = "Performance Metrics of Algorithms",
       x = "Performance Metric",
       y = "Value") +
  theme_minimal() +
  theme(axis.text.x = element_text(angle = 45, hjust = 1))

# Print the plot
